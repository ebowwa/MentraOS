# CMake configuration for I2S Simple Loopback Test
# This builds only the I2S loopback functionality without LC3 or BLE

cmake_minimum_required(VERSION 3.20.0)
find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})
project(nrf5340_i2s_loopback_test)

# Use our simple I2S loopback main function
target_sources(app PRIVATE 
    src/main_i2s_loopback.c
    src/i2s_simple_loopback.c
)

# Include header directory
target_include_directories(app PRIVATE src/)
